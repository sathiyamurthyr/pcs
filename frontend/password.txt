(function() {
    // Detect environment
    const hostname = window.location.hostname;
    let apiUrl;
    
    if (hostname === 'localhost' || hostname === '127.0.0.1') {
        // Local development
        apiUrl = 'http://localhost:3000/api';
    } else {
        // Production - change 'yourdomain.com' to your actual domain
        apiUrl = `https://${hostname}/api`;
        // Or hardcode: apiUrl = 'https://yourdomain.com/api';
    }
    
    window.API_URL = apiUrl;
    console.log('API URL:', window.API_URL);
})();





const express = require('express');
const cors = require('cors');
const path = require('path');
const dotenv = require('dotenv');

dotenv.config();

const app = express();

// CORS configuration - allow all domains (or restrict to your domain)
app.use(cors({
    origin: '*', // For production, replace with your domain
    methods: ['GET', 'POST', 'PUT', 'DELETE', 'OPTIONS'],
    allowedHeaders: ['Content-Type', 'Authorization']
}));

app.use(express.json());
app.use(express.urlencoded({ extended: true }));

// Serve static files
app.use('/uploads', express.static('uploads'));
app.use(express.static('frontend'));

// Routes
const authRoutes = require('./routes/auth');
const adminRoutes = require('./routes/admin');
const memberRoutes = require('./routes/members');
const appointmentRoutes = require('./routes/appointments');
const contactRoutes = require('./routes/contact');
const galleryRoutes = require('./routes/gallery');
const enquiryRoutes = require('./routes/enquiries');

app.use('/api/auth', authRoutes);
app.use('/api/admin', adminRoutes);
app.use('/api/members', memberRoutes);
app.use('/api/appointments', appointmentRoutes);
app.use('/api/contact', contactRoutes);
app.use('/api/gallery', galleryRoutes);
app.use('/api/enquiries', enquiryRoutes);

// Serve index.html for all other routes (SPA support)
app.get('*', (req, res) => {
    res.sendFile(path.join(__dirname, 'frontend', 'index.html'));
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => {
    console.log(`Server running on port ${PORT}`);
});
